CREATE KEYSPACE proyecto
  WITH REPLICATION = { 
   'class' : 'SimpleStrategy', 
   'replication_factor' : 1 
  };

use proyecto;

create table jugadores(
id_jugador int primary key,
fecha_nac date,
nombre_jugador text,
peso_base int,
peso_extra int);

id_jugador,fecha_nac,nombre_jugador,peso_base,peso_extra

create table entrenador(
id_entrenador int primary key,
fecha_nac date,
nombre_entrenador text,
peso_base int,
peso_extra int);

create table arbitro(
id_arbitro int primary key,
fecha_nac date,
nombre_arbitro text,
peso_base int,
peso_extra int);

create table clubes(
id_club int primary key,
nombre_club text);

create table equipo(
id_equipo int primary key,
nombre_equipo text
id_entrenador int);

create table quiniela(
id_quiniela int primary key,
fecha_inicio date,
fecha_fin date,
inversion int,
ganancia int);

create table predicciones(
id_prediccion int primary key,
id_partido int,
monto int,
goles_local int,
goles_visita int);

create table partido(
id_encuentro int primary key,
id_equipo_local int,
id_equipo_visita int,
id_arbitro int,
id_arbitro_auxiliar1 int
id_arbitro_auxiliar2 int,
goles_local int,
goles_visita int,
fecha_juego date);

create table equipo_jugador(
id_equipo int,
id_jugador int primary key,
reserva boolean);

create table club_jugador(
id_club int,
id_jugador int primary key);
